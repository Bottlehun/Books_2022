## 1. 본문 드래그

확인버튼이 없음<br/>
드롭시 정답이면 정답효과음, 오답이면 다시 제자리로<br/>
정답이면 드롭 드래그 모두 사라짐.


* ### 1.1 XHTML 설정
    ```html
    <!-- 드래그 앤 드롭 -->
    var dragItems = {
        drop: ['1', '2', '3'],
        drag: ['1', '2', '3'],
    }
    dragCon = new dragContents($(".dragdrop-group-cont"), {
        items: dragItems,
        answerEvent: function($this){
            // console.log("dragItemEndEvent : "+$this);
            모두 정답을 맞췄을 경우 실행될 명령어
        },
        dragItemEndEvent: function($this){
            // console.log("dropItemEndEvent : "+$this);
            드래그 한번 맞추면 실행될 명령어
        },
    });
    <!-- 드래그 앤 드롭 -->
    ```
